CREATE TABLE `user_information` (  `USER_ID` bigint(10) NOT NULL AUTO_INCREMENT,  `USER_NAME` varchar(20) DEFAULT NULL COMMENT '用户名',  `USER_PASSWORD` varchar(20) DEFAULT NULL COMMENT '密码',  `USER_SEX` varchar(10) DEFAULT NULL COMMENT '性别',  `USER_AGE` int(10) DEFAULT NULL COMMENT '年龄',  `USER_EMAIL` varchar(20) DEFAULT NULL COMMENT 'email',  PRIMARY KEY (`USER_ID`)) ENGINE=InnoDB AUTO_INCREMENT=25 DEFAULT CHARSET=utf8
